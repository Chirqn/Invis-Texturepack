    function updateHatImagesOnce(observer) {
        let allReplaced = true;

        for (const [index, url] of Object.entries(hatImageMap)) {
            const img = document.querySelector(`#hat_menu_content > div:nth-child(${index}) > img`);
            if (img) {
                if (!img.dataset.replaced) {
                    img.src = url;
                    img.dataset.replaced = "true"; // mark as replaced
                }
            } else {
                allReplaced = false; // missing one, keep watching
            }
        }

        // Stop watching once all hats are replaced
        if (allReplaced) {
            observer.disconnect();
        }
    }

    // Create observer
    const observer = new MutationObserver(() => updateHatImagesOnce(observer));
    observer.observe(document.body, { childList: true, subtree: true });

    function redirectIfMatched(url) {
        // Create an array from Map entries and sort by length of match descending
        const sortedEntries = Array.from(redirects.entries())
        .sort((a, b) => b[0].length - a[0].length);

        for (const [match, replacement] of sortedEntries) {
            if (url.includes(match)) {
                return replacement;
            }
        }
        return url;
    }

    // Patch HTMLImageElement.prototype.src
    function patchImageSrc() {
        const proto = HTMLImageElement.prototype;
        const desc = Object.getOwnPropertyDescriptor(proto, 'src');
        if (!desc || !desc.set) return;

        const originalSetter = desc.set;

        Object.defineProperty(proto, 'src', {
            configurable: true,
            enumerable: true,
            get: desc.get,
            set: function(value) {
                const redirected = redirectIfMatched(value);
                if (redirected !== value) {
                }
                return originalSetter.call(this, redirected);
            }
        });
        console.log("[Redirector] Patched img.src");
    }

    // Patch setAttribute
    const originalSetAttr = Element.prototype.setAttribute;
    Element.prototype.setAttribute = function(name, value) {
        if (this.tagName === 'IMG' && name === 'src') {
            const redirected = redirectIfMatched(value);
            if (redirected !== value) {
            }
            value = redirected;
        }
        return originalSetAttr.call(this, name, value);
    };
    console.log("[Redirector] Patched setAttribute for IMG.src");

    // Patch new Image()
    const OriginalImage = window.Image;
    window.Image = new Proxy(OriginalImage, {
        construct(target, args) {
            const img = new target(...args);

            // Patch setAttribute on the new image
            const origSetAttr = img.setAttribute;
            img.setAttribute = function(name, value) {
                if (name === "src") {
                    const redirected = redirectIfMatched(value);
                    if (redirected !== value) {

                return origSetAttr.call(this, name, value);
            };

            // Patch src property setter/getter on the new image

                    const redirected = redirectIfMatched(value);
                    if (redirected !== value) {

                    return HTMLImageElement.prototype.src.set.call(this, redirected);
                },
                get() {
                    return HTMLImageElement.prototype.src.get.call(this);
                },
                configurable: true
            });

            return img;
        }
    });
